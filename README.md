> on:
>   schedule: [{ cron: "0 0 * * *" }]
>   workflow_dispatch:
> jobs:
>   snake:
>     runs-on: ubuntu-latest
>     steps:
>       - uses: Platane/snk@v3

>           github_user_name: 0xMayurrr
